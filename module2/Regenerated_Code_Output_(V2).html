import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# Filter Q1 2024 and Q1 2025
df_q1_2024 = df[(df['year'] == 2024) & (df['quarter'] == 1)]
df_q1_2025 = df[(df['year'] == 2025) & (df['quarter'] == 1)]

# Group by coffee_name and sum the price
sales_2024 = df_q1_2024.groupby('coffee_name')['price'].sum().reset_index()
sales_2025 = df_q1_2025.groupby('coffee_name')['price'].sum().reset_index()

# Merge the two DataFrames for comparison and fill missing values with 0
sales_comparison = pd.merge(sales_2024, sales_2025, on='coffee_name', how='outer', suffixes=('_2024', '_2025')).fillna(0)

# Sort by coffee_name for consistent ordering
sales_comparison = sales_comparison.sort_values('coffee_name')

# Prepare bar positions
labels = sales_comparison['coffee_name']
x = np.arange(len(labels))
width = 0.35

plt.figure(figsize=(10, 6))
bars2024 = plt.bar(x - width/2, sales_comparison['price_2024'], width, label='Q1 2024', color='skyblue')
bars2025 = plt.bar(x + width/2, sales_comparison['price_2025'], width, label='Q1 2025', color='salmon')

# Title and axis labels
plt.title('Q1 Coffee Sales Comparison: 2024 vs 2025')
plt.xlabel('Coffee Name')
plt.ylabel('Total Sales Price')
plt.xticks(x, labels, rotation=45, ha='right')
plt.legend()
plt.grid(axis='y', linestyle='--', alpha=0.7)

# Annotate bar values
max_height = max(sales_comparison['price_2024'].max(), sales_comparison['price_2025'].max())
offset = max_height * 0.01
for bar in bars2024:
    h = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2, h + offset, f'{h:.0f}', ha='center', va='bottom', fontsize=8)
for bar in bars2025:
    h = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2, h + offset, f'{h:.0f}', ha='center', va='bottom', fontsize=8)

plt.tight_layout()
plt.savefig('chart_v2.png', dpi=300)
plt.close()
