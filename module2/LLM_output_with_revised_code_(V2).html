import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Filter for Q1 of each year
q1_2024 = df[(df['year'] == 2024) & (df['quarter'] == 1)]
q1_2025 = df[(df['year'] == 2025) & (df['quarter'] == 1)]

# Aggregate total sales by coffee_name
sales_2024 = q1_2024.groupby('coffee_name')['price'].sum().reset_index().rename(columns={'price': 'sales_2024'})
sales_2025 = q1_2025.groupby('coffee_name')['price'].sum().reset_index().rename(columns={'price': 'sales_2025'})

# Merge and fill missing with zero
sales = pd.merge(sales_2024, sales_2025, on='coffee_name', how='outer').fillna(0)

# Sort by coffee_name
sales = sales.sort_values('coffee_name')

# Prepare bar positions
names = sales['coffee_name'].tolist()
x = np.arange(len(names))
width = 0.35

# Plot grouped bars
plt.figure(figsize=(10, 6))
plt.bar(x - width/2, sales['sales_2024'], width, label='2024', color='#4C72B0')
plt.bar(x + width/2, sales['sales_2025'], width, label='2025', color='#DD8452')

# Labels and title
plt.xlabel('Coffee Name')
plt.ylabel('Total Sales Price')
plt.title('Q1 Coffee Sales Comparison (2024 vs 2025)')
plt.xticks(x, names, rotation=45, ha='right')
plt.legend()
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.tight_layout()

# Save and close
plt.savefig('drink_sales_v2.png', dpi=300)
plt.close()
